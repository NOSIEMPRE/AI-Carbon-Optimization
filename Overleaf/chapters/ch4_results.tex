\subsection{Experimental Setup}

\textbf{Baselines.}
\begin{enumerate}[leftmargin=1.5em]
  \item \textbf{Carbon-Agnostic}: load distributed uniformly across regions and hours, ignoring CI.
  \item \textbf{Greedy Temporal}: flexible load pushed to lowest-CI hours within each region; no spatial routing.
  \item \textbf{LP (this thesis)}: joint spatio-temporal LP with RF weighting ($\alpha$ tuned via sensitivity analysis).
  \item \textbf{Oracle}: LP with perfect future CI/RF knowledge (theoretical upper bound).
\end{enumerate}

\textbf{Metrics.}
\begin{itemize}[leftmargin=1.5em]
  \item Carbon savings (\%) relative to Carbon-Agnostic baseline.
  \item Optimality gap (\%) of LP vs.\ Oracle.
  \item Temporal vs.\ spatial decomposition: marginal savings from each dimension via counterfactual solves.
\end{itemize}
