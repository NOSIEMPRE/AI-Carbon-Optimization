\section{Supplementary Figures}
\label{app:supp}

The following figures support the exploratory analysis of Section~\ref{sec:problem} and the results of Section~\ref{sec:results}. They are referenced from the main text where relevant and collected here to keep the body within the page limit.

\begin{figure}[h]
\centering
\includegraphics[width=\linewidth]{eda_diurnal}
\caption{Average carbon intensity by hour of day (UTC) per region. The intra-day range (shown in the legend) is largest for Belgium, driven by midday solar, and smallest for Finland, whose carbon intensity is nearly flat across the day despite being the most variable region overall on a multi-day basis.}
\label{fig:eda-diurnal}
\end{figure}

\begin{figure}[h]
\centering
\includegraphics[width=0.82\linewidth]{eda_cfe_composition}
\caption{Mean composition of carbon-free supply by region, split into renewable (wind, solar, hydro) and nuclear shares. PJM and NYISO reach much of their carbon-free fraction through nuclear rather than renewables, which is why a renewables-only signal would misrepresent their cleanliness and the carbon-free energy fraction is used instead.}
\label{fig:eda-comp}
\end{figure}

\begin{figure}[h]
\centering
\includegraphics[width=\linewidth]{perturbation_test}
\caption{Hourly schedules for three of the perturbation scenarios in Table~\ref{tab:pert} (the CI shock is reported in the table but not plotted). Demand increments flow to the cleanest region (left), a shorter deferral window forces load out of the clean hours and raises carbon (centre), and a tighter hourly cap first spreads Finland's load and then reduces it (right).}
\label{fig:f-pert}
\end{figure}

\begin{figure}[h]
\centering
\includegraphics[width=\linewidth]{schedule_sample}
\caption{No-shift baseline versus the LP-optimized schedule for one representative window, with the carbon-intensity signal below. The LP routes the batch to Finland and spreads it across hours to satisfy the ramp limit, cutting window carbon from 533 to 53~gCO\textsubscript{2}.}
\label{fig:f-sched}
\end{figure}

\begin{figure}[h]
\centering
\includegraphics[width=0.92\linewidth]{cv_regression}
\caption{Within-region carbon saving from timing alone against the coefficient of variation of carbon intensity, one point per region. Timing helps materially only in the high-variability grids (Belgium, Finland) and barely at all in the flat ones (PJM, NYISO, Singapore).}
\label{fig:f-cv}
\end{figure}
